    <script>
        // Dữ liệu học tập mẫu
        const levelData = {
            don: { hanzi: '苹', pinyin: 'píng', meaning: 'Táo (Luyện âm Píng)', match: '苹' },
            phuc: { hanzi: '苹果', pinyin: 'píng guǒ', meaning: 'Quả Táo', match: '苹果' },
            cau: { hanzi: '我想吃苹果', pinyin: 'wǒ xiǎng chī píng guǒ', meaning: 'Con muốn ăn táo.', match: '我想吃苹果' }
        };

        let currentLevel = 'phuc';
        let score = 0;
        let isListening = false;

        // Cấu hình Nhận diện giọng nói Web Speech API
        const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
        let recognition;

        if (SpeechRecognition) {
            recognition = new SpeechRecognition();
            recognition.lang = 'zh-CN'; // Nhận diện tiếng Trung Phổ Thông
            recognition.interimResults = false;
            recognition.maxAlternatives = 1;

            recognition.onresult = (event) => {
                const babySpeechResult = event.results[0][0].transcript;
                document.getElementById('result-display').innerText = `Bé vừa nói: "${babySpeechResult}"`;
                
                const target = levelData[currentLevel].match;
                // Kiểm tra xem giọng đọc của bé có khớp với từ mục tiêu không
                if (babySpeechResult.includes(target) || target.includes(babySpeechResult)) {
                    score += 10;
                    document.getElementById('score-board').innerText = `⭐ ${score} Điểm`;
                    document.getElementById('panda-msg').innerText = "Oa! Bé phát âm chuẩn quá! Khu vườn được cộng thêm 10 sao thưởng! 🌸🎉";
                } else {
                    document.getElementById('panda-msg').innerText = "Gần đúng rồi, bé ấn nút nói lại rõ ràng hơn thử xem sao nhé! 💪";
                }
            };

            recognition.onerror = (event) => {
                console.error(event.error);
                stopListening();
            };

            recognition.onend = () => {
                stopListening();
            };
        } else {
            alert("Trình duyệt này không hỗ trợ tính năng nhận diện giọng nói. Bạn hãy thử dùng Google Chrome nhé!");
        }

        // Hàm Đổi Cấp Độ (Từ đơn / Từ phức / Câu ngắn)
        function changeLevel(level) {
            currentLevel = level;
            document.getElementById('pinyin-display').innerText = levelData[level].pinyin;
            document.getElementById('hanzi-display').innerText = levelData[level].hanzi;
            document.getElementById('meaning-display').innerText = `(${levelData[level].meaning})`;
            document.getElementById('result-display').innerText = "";
            document.getElementById('panda-msg').innerText = "Hạt mầm mới đã sẵn sàng! Ấn chiếc loa xanh để nghe cô phát âm mẫu nhé!";
        }

        // Hàm Phát Âm Mẫu (Text-to-Speech)
        function speakSample() {
            if ('speechSynthesis' in window) {
                const text = levelData[currentLevel].hanzi;
                const utterance = new SpeechSynthesisUtterance(text);
                utterance.lang = 'zh-CN'; // Tiếng Trung
                utterance.rate = 0.6;     // Đọc chậm rãi cho trẻ em dễ bắt chước
                utterance.pitch = 1.3;    // Giọng cao, trong trẻo, dễ thương
                
                document.getElementById('panda-msg').innerText = "Bé chú ý lắng nghe cô đọc mẫu này... 🎧";
                window.speechSynthesis.speak(utterance);
            } else {
                alert("Trình duyệt của bạn không hỗ trợ tính năng phát âm mẫu.");
            }
        }

        // Hàm bật/tắt ghi âm giọng của bé
        function toggleListening() {
            if (!recognition) return;
            if (!isListening) {
                isListening = true;
                document.getElementById('mic-button').classList.add('listening');
                document.getElementById('panda-msg').innerText = "Cô đang nghe đây, bé đọc thật to lên nào! 🎙️";
                document.getElementById('result-display').innerText = "Hệ thống đang nghe...";
                recognition.start();
            } else {
                stopListening();
                recognition.stop();
            }
        }

        function stopListening() {
            isListening = false;
            document.getElementById('mic-button').classList.remove('listening');
        }
    </script>
